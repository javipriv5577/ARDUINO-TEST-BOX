#include <Wire.h> 
#include <LiquidCrystal_I2C.h>

//Crear el objeto lcd  direcci√≥n  0x3F y 16 columnas x 2 filas
LiquidCrystal_I2C lcd(0x20,16,2);  

void setup() {
  	//Inicializar el LCD
  	lcd.init();
  
  	//Encender la luz de fondo
  	lcd.backlight();
  
  	//Mensaje en el LCD
  	lcd.print("Hola");
   	lcd.setCursor(0,1);
	delay(10);
}

void loop() 
{
  	lcd.setCursor(0,1);
  	lcd.print("ARDUINO TEST BOX");
  	delay(10);
}
